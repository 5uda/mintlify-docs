---
title: "Criar Pagamento PIX"
api: "POST /payment/create"
description: "Cria uma nova transação PIX e gera o QR Code e o código Copia e Cola."
---

### Parâmetros do Corpo (JSON)

<ParamField body="value" type="number" required>
  Valor em reais (ex: 100.50). Mínimo R$ 1,00.
</ParamField>

<ParamField body="description" type="string">
  Descrição que aparecerá no extrato.
</ParamField>

<ParamField body="coverFee" type="boolean" default="false">
  Se `true`, a taxa é adicionada ao valor total. Se `false`, a taxa é descontada do valor recebido.
</ParamField>

<ParamField body="splitUser" type="string">
  Email de outro usuário Zynx para divisão de valores.
</ParamField>

<ParamField body="splitTax" type="number">
  Porcentagem (0.01 a 100) para o `splitUser`.
</ParamField>

### Resposta de Sucesso (201 Created)

<ResponseField name="success" type="boolean">
  Indica se a operação foi bem-sucedida.
</ResponseField>

<ResponseField name="data" type="object">
  <Expandable title="Propriedades do Objeto">
    <ResponseField name="id" type="string">
      ID único da transação.
    </ResponseField>
    <ResponseField name="transactionId" type="string">
      ID interno da transação.
    </ResponseField>
    <ResponseField name="valueInReais" type="number">
      Valor total em reais.
    </ResponseField>
    <ResponseField name="netValue" type="number">
      Valor líquido em centavos.
    </ResponseField>
    <ResponseField name="fee" type="number">
      Taxa aplicada em centavos.
    </ResponseField>
    <ResponseField name="status" type="string">
      Status atual (`PENDING`).
    </ResponseField>
    <ResponseField name="qrcodeUrl" type="string">
      URL ou Base64 da imagem do QR Code.
    </ResponseField>
    <ResponseField name="copyPaste" type="string">
      Código PIX Copia e Cola.
    </ResponseField>
    <ResponseField name="createdAt" type="string">
      Data de criação da transação.
    </ResponseField>
  </Expandable>
</ResponseField>

<RequestExample>
```bash cURL
curl -X POST 'https://api.zynxwallet.com.br/api/v1/payment/create' \
  -H 'X-API-Key: zynx_sua_chave_aqui' \
  -H 'Content-Type: application/json' \
  -d '{
    "value": 150.00,
    "description": "Venda de Produto #123",
    "coverFee": true,
    "splitUser": "parceiro@zynx.com",
    "splitTax": 10.0
  }'
```
</RequestExample>
