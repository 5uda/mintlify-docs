---
title: "Consultar Status de Pagamento"
api: "GET /payment/get/{id}"
description: "Verifica os detalhes e o status atual de um pagamento específico."
---

### Path Parameters

<ParamField path="id" type="string" required>
  O ID da transação (ex: `pay_abc123`)
</ParamField>

### Resposta de Sucesso (200 OK)

<ResponseField name="success" type="boolean">
  Indica se a operação foi bem-sucedida.
</ResponseField>

<ResponseField name="data" type="object">
  <Expandable title="Propriedades do Objeto">
    <ResponseField name="id" type="string">
      ID único da transação.
    </ResponseField>
    <ResponseField name="status" type="string">
      Status atual (`PENDING`, `COMPLETED`, `FAILED`).
    </ResponseField>
    <ResponseField name="value" type="number">
      Valor total em centavos.
    </ResponseField>
    <ResponseField name="netValue" type="number">
      Valor líquido em centavos.
    </ResponseField>
    <ResponseField name="updatedAt" type="string">
      Data da última atualização.
    </ResponseField>
  </Expandable>
</ResponseField>

<RequestExample>
```bash cURL
curl -X GET 'https://api.zynxwallet.com.br/api/v1/payment/get/pay_abc123' \
  -H 'X-API-Key: zynx_sua_chave_aqui'
```
</RequestExample>

<ResponseExample>
```json Resposta
{
  "success": true,
  "data": {
    "id": "pay_abc123",
    "status": "COMPLETED",
    "value": 15300,
    "netValue": 15000,
    "updatedAt": "2024-02-16T15:05:00Z"
  }
}
```
</ResponseExample>
